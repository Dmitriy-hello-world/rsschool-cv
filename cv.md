## Code example
Decimal to binary converter:  
```function decToBin(num) {
  if (num == 0) {
    return '0';
  }
  let binary = 0,
      rem,
      i = 1;
  while (num != 0) {
      rem = num % 2;
      num = parseInt(num / 2);
      binary += (rem * i);
      i = i * 10;
  }
  return `${binary}`;
};
```
